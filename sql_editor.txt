create table employer_profiles (
  id uuid default gen_random_uuid() primary key,

  user_id uuid not null unique
    references profiles(id) on delete cascade,

  gender text not null
    check (gender in ('male', 'female', 'other')),

  -- 👤 Full Name (2–50, alphabets + spaces)
  full_name text not null
    check (
      char_length(full_name) between 2 and 50
      and full_name ~ '^[A-Za-z ]+$'
    ),

  -- 🏢 Business Name (2–100, allowed chars)
  business_name text not null
    check (
      char_length(business_name) between 2 and 100
      and business_name ~ '^[A-Za-z0-9 .&-]+$'
    ),

  -- 📍 Location (2–100)
  business_location text not null
    check (
      char_length(business_location) between 2 and 100
      and business_location ~ '^[A-Za-z0-9, ]+$'
    ),

  latitude float,
  longitude float,

  -- 📌 Pincode (exactly 6 digits)
  pincode char(6) not null
    check (pincode ~ '^[0-9]{6}$'),

  -- 📧 Email (nullable if user doesn't have one)
  business_email text
    check (
      business_email is null
      or business_email ~ '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'
    ),

  has_business_email boolean not null default true,

  is_verified boolean default false,

  created_at timestamptz default now()
);
